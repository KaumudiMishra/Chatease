require('dotenv').config();
const express = require('express');
const bodyParser = require('body-parser');
const { MessagingResponse } = require('twilio').twiml;
const admin = require('firebase-admin');
const axios = require('axios');                 // ← NEW: HTTP client for OpenRouter

const app = express();
app.use(bodyParser.urlencoded({ extended: false }));

// ── Firebase Admin ───────────────────────────────────────────────
const serviceAccount = require('./serviceAccountKey.json');

admin.initializeApp({
  credential: admin.credential.cert(serviceAccount),
  databaseURL: 'https://nwbot-c2fc9-default-rtdb.asia-southeast1.firebasedatabase.app',
});

const db = admin.database();

// ── Helper: talk to OpenRouter LLM ───────────────────────────────
async function getAIReply(userPrompt) {
  try {
    const { data } = await axios.post(
      'https://openrouter.ai/api/v1/chat/completions',
      {
        model: 'openai/gpt-3.5-turbo',
        messages: [
          {
            role: 'system',
            content:
              'You are a concise, polite college information assistant. Reply in 2–3 sentences.',
          },
          { role: 'user', content: userPrompt },
        ],
        max_tokens: 300,
      },
      {
        headers: {
          Authorization: `Bearer ${process.env.OPENROUTER_API_KEY}`,
          'Content-Type': 'application/json',
          // Optionally add these if required for free keys:
          // 'HTTP-Referer': 'https://your‑domain.com',
          // 'X-Title': 'CollegeWhatsAppBot',
        },
      }
    );

    return data.choices[0].message.content.trim();
  } catch (err) {
    console.error('❌ OpenRouter error:', err.message);
    return "⚠️ I'm having trouble answering that right now. Please try again later.";
  }
}

// ── Incoming WhatsApp webhook ────────────────────────────────────
app.post('/incoming', async (req, res) => {
  const userMsg = (req.body.Body || '').toLowerCase().trim();
  console.log('📩 Incoming:', userMsg);

  const twiml = new MessagingResponse();

  try {
    // 1) Try exact key in Firebase
    const snap = await db.ref(userMsg).once('value');

    if (snap.exists()) {
      twiml.message(snap.val()); // Found in FAQ DB
    } else {
      // 2) Fall back to OpenRouter GPT
      const aiReply = await getAIReply(userMsg);
      twiml.message(aiReply);
    }
  } catch (fbErr) {
    console.error('❌ Firebase error:', fbErr);
    twiml.message('⚠️ Database error. Please try again later.');
  }

  res.type('text/xml').send(twiml.toString());
});

// ── Start server ────────────────────────────────────────────────
app.listen(3000, () => {
  console.log('🚀 Bot server is running on http://localhost:3000');
});
