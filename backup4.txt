require('dotenv').config();
const express = require('express');
const bodyParser = require('body-parser');
const { MessagingResponse } = require('twilio').twiml;
const admin = require('firebase-admin');
const axios = require('axios');
const fs = require('fs');
const Fuse = require('fuse.js');

const app = express();
app.use(bodyParser.urlencoded({ extended: false }));

// ─── Firebase Setup ────────────────────────────────────────
const serviceAccount = require('./serviceAccountKey.json');
admin.initializeApp({
  credential: admin.credential.cert(serviceAccount),
  databaseURL: process.env.FIREBASE_DB_URL,
});
const db = admin.firestore();

// ─── Load FAQ Data ─────────────────────────────────────────
const faq = require('./faq.json');

function extractAllKeys(obj, prefix = '') {
  return Object.keys(obj).flatMap(key => {
    const fullKey = prefix ? `${prefix}.${key}` : key;
    return typeof obj[key] === 'object' && !Array.isArray(obj[key])
      ? extractAllKeys(obj[key], fullKey)
      : fullKey;
  });
}
const VALID_KEYS = extractAllKeys(faq);

function getNestedAnswer(obj, path) {
  const result = path.split('.').reduce((o, key) => (o && o[key] ? o[key] : null), obj);
  return typeof result === 'string' ? result : null;
}

// ─── Learned Phrases ───────────────────────────────────────
const learnedPhrases = [];
if (fs.existsSync('./learned_intents.jsonl')) {
  fs.readFileSync('./learned_intents.jsonl', 'utf8')
    .split('\n')
    .filter(Boolean)
    .forEach(line => {
      try {
        const item = JSON.parse(line);
        learnedPhrases.push({ phrase: item.user.toLowerCase(), intent: item.intent });
      } catch (e) {
        console.error("❌ Error parsing learned intent line:", e.message);
      }
    });
}

function fuzzyLearnedMatch(userMsg) {
  const fuse = new Fuse(learnedPhrases, {
    keys: ['phrase'],
    includeScore: true,
    threshold: 0.33,
  });
  const results = fuse.search(userMsg.toLowerCase());
  if (results.length > 0) {
    console.log(`🔍 Learned Match: ${results[0].item.intent} | Score: ${results[0].score}`);
    return results[0].item.intent;
  }
  return null;
}

function fuzzyKeyMatch(userMsg) {
  const fuse = new Fuse(VALID_KEYS, {
    includeScore: true,
    threshold: 0.33,
  });
  const results = fuse.search(userMsg.toLowerCase());
  if (results.length > 0) {
    console.log(`🔍 Key Match: ${results[0].item} | Score: ${results[0].score}`);
    return results[0].item;
  }
  return null;
}

async function classifyIntent(userInput) {
  const prompt = `
You are an intent-mapping assistant. Your job is to match user input to one of the following keys exactly.

Valid Keys:
${VALID_KEYS.join('\n')}

User Input: "${userInput}"
Which key does this best match? Respond with ONLY the key or "unknown".`.trim();

  try {
    const { data } = await axios.post(
      'https://openrouter.ai/api/v1/chat/completions',
      {
        model: 'mistralai/mistral-7b-instruct',
        messages: [{ role: 'user', content: prompt }],
        max_tokens: 20,
        temperature: 0.1,
      },
      {
        headers: {
          Authorization: `Bearer ${process.env.OPENROUTER_API_KEY}`,
          'Content-Type': 'application/json',
        },
      }
    );

    let raw = data.choices[0].message.content.trim().toLowerCase();
    if (!VALID_KEYS.includes(raw)) {
      const match = VALID_KEYS.find(k => k.startsWith(raw) || raw.includes(k));
      return match || 'unknown';
    }
    return raw;
  } catch (err) {
    console.error('❌ GPT classification error:', err.message);
    return 'unknown';
  }
}

app.post('/incoming', async (req, res) => {
  const twiml = new MessagingResponse();
  const userMsg = req.body.Body.trim();

  console.log(`📩 Incoming message: "${userMsg}"`);

  let intent = fuzzyLearnedMatch(userMsg);

  if (!intent) intent = fuzzyKeyMatch(userMsg);

  if (!intent) intent = await classifyIntent(userMsg);
  console.log(`🧠 Intent Detected: ${intent}`);

  let response = getNestedAnswer(faq, intent);

  if (!response) {
    console.log("⚠ No match found. Using GPT fallback restricted to PIET context.");
    try {
      const prompt = `
You are a professional chatbot for "Poornima Institute of Engineering and Technology (PIET), Jaipur".

STRICT INSTRUCTIONS:
- DO NOT answer any question that is NOT about PIET (no fun facts, science, psychology, history, jokes, movies, religion, comparisons).
- ONLY reply if the question is about PIET's admission, fees, hostel, placements, campus life, etc.
- If a user asks anything unrelated to PIET, simply respond:
"I'm designed to assist with Poornima Institute-related queries only."

User Query: "${userMsg}"
Response:`.trim();

      const { data } = await axios.post(
        'https://openrouter.ai/api/v1/chat/completions',
        {
          model: 'mistralai/mistral-7b-instruct',
          messages: [{ role: 'user', content: prompt }],
          max_tokens: 150,
          temperature: 0.5,
        },
        {
          headers: {
            Authorization: `Bearer ${process.env.OPENROUTER_API_KEY}`,
            'Content-Type': 'application/json',
          },
        }
      );

      response = data.choices[0].message.content.trim();
      console.log(`💡 GPT Fallback Response: ${response}`);

      fs.appendFileSync(
        'fallback_log.txt',
        `\n[${new Date().toISOString()}]\nUser: ${userMsg}\nIntent: ${intent}\nGPT: ${response}\n---\n`
      );
      fs.appendFileSync(
        'learned_intents.jsonl',
        JSON.stringify({ user: userMsg, intent: intent, answer: response }) + "\n"
      );
      learnedPhrases.push({ phrase: userMsg.toLowerCase(), intent });
    } catch (err) {
      console.error("❌ GPT fallback error:", err.message);
      response = "Sorry, I'm having trouble fetching that information. Please try again later.";
    }
  }

  console.log(`📤 Bot reply: "${response}"`);
  twiml.message(response);
  res.writeHead(200, { 'Content-Type': 'text/xml' });
  res.end(twiml.toString());
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  console.log(`🚀 Server running on port ${PORT}`);
});
